import requests
url = 'https://api.quantdata.io/query'
params = {
'function': 'TIME_SERIES_DAILY',
'symbol': 'IBM',
'apikey': 'YOUR_KEY'
}
r = requests.get(url, params=params)
print(r.json())
const axios = require('axios');
const url = 'https://api.quantdata.io/query';
const params = {
function: 'TIME_SERIES_DAILY',
symbol: 'IBM',
apikey: 'YOUR_KEY'
};
axios.get(url, { params })
.then(res => console.log(res.data));
